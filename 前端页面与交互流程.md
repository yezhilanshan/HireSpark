```typescript
// interview/page.tsx
'use client'

import { useEffect, useState, useRef } from 'react'
import { useRouter } from 'next/navigation'
import { SocketClient } from '@/lib/socket'

export default function InterviewPage() {
  const router = useRouter()
  const socketRef = useRef<SocketClient | null>(null)
  
  const [currentQuestion, setCurrentQuestion] = useState('')
  const [userAnswer, setUserAnswer] = useState('')
  const [chatHistory, setChatHistory] = useState<any[]>([])
  const [isLoading, setIsLoading] = useState(false)
  const [config, setConfig] = useState(null)
  
  // 初始化
  useEffect(() => {
    // 1️⃣ 从 sessionStorage 获取配置
    const savedConfig = sessionStorage.getItem('interviewConfig')
    const { round, position, difficulty } = JSON.parse(savedConfig)
    setConfig({ round, position, difficulty })
    
    // 2️⃣ 初始化 Socket
    socketRef.current = SocketClient.getInstance()
    
    // 3️⃣ 建立连接
    socketRef.current.connect().then(() => {
      // 4️⃣ 监听事件
      socketRef.current.on('interview_started', (data) => {
        console.log('✓ 面试已启动')
        setCurrentQuestion(data.initial_question)
      })
      
      socketRef.current.on('llm_response', (data) => {
        console.log('✓ 收到 LLM 回复')
        // 添加到对话历史
        setChatHistory(prev => [...prev, {
          question: currentQuestion,
          answer: userAnswer,
          feedback: data.feedback,
          timestamp: new Date()
        }])
        setUserAnswer('')
        setIsLoading(false)
      })
      
      socketRef.current.on('interview_ended', (data) => {
        console.log('✓ 面试已结束，跳转到报告')
        // 跳转到报告页
        router.push(`/report?reportPath=${data.report_path}`)
      })
      
      socketRef.current.on('error', (data) => {
        console.error('❌ 错误:', data.error)
        alert(`错误: ${data.error}`)
      })
      
      // 5️⃣ 启动面试
      socketRef.current.emit('start_interview', {
        round,
        position,
        difficulty,
        user_id: 'default'
      })
    })
    
    return () => {
      // 清理资源
    }
  }, [])
  
  // 提交答案
  const handleSubmitAnswer = async () => {
    if (!userAnswer.trim()) {
      alert('请输入答案')
      return
    }
    
    setIsLoading(true)
    
    // 发送到后端处理
    socketRef.current.emit('llm_process_answer', {
      user_answer: userAnswer,
      current_question: currentQuestion,
      position: config.position,
      chat_history: chatHistory
    })
  }
  
  // 结束面试
  const handleEndInterview = () => {
    if (confirm('确定要结束面试吗？')) {
      socketRef.current.emit('end_interview', {})
    }
  }
  
  return (
    <div className="flex h-screen">
      {/* 左侧: 摄像头和实时数据 */}
      <div className="w-1/3 p-4 bg-gray-900 text-white">
        <section>
          <h3>📹 摄像头预览</h3>
          <video 
            autoPlay 
            playsInline 
            className="w-full rounded-lg"
          />
        </section>
        
        <section className="mt-4">
          <h3>📊 实时检测数据</h3>
          <div>😊 表情: 专注 85%</div>
          <div>👁️ 眼神: 正常</div>
          <div>🎤 声音: 清晰</div>
        </section>
      </div>
      
      {/* 右侧: 问题、对话、输入 */}
      <div className="w-2/3 p-8 flex flex-col">
        {/* 问题显示 */}
        <section className="mb-6 p-4 border rounded-lg">
          <h3 className="font-bold">📝 当前问题:</h3>
          <p className="text-lg mt-2">{currentQuestion}</p>
          <p className="text-sm text-gray-500 mt-2">⏱️ 已用时间: 1:23</p>
        </section>
        
        {/* 对话历史 */}
        <section className="flex-1 mb-6 p-4 border rounded-lg overflow-y-auto bg-gray-50">
          <h3 className="font-bold">💬 对话历史:</h3>
          {chatHistory.map((item, idx) => (
            <div key={idx} className="mt-4 border-t pt-2">
              <div><strong>Q{idx + 1}:</strong> {item.question}</div>
              <div><strong>A{idx + 1}:</strong> {item.answer}</div>
              <div className="text-blue-600">
                <strong>[LLM反馈]:</strong> {item.feedback}
              </div>
            </div>
          ))}
        </section>
        
        {/* 答案输入 */}
        <section className="mb-6">
          <h3 className="font-bold mb-2">📝 输入你的回答:</h3>
          <textarea
            value={userAnswer}
            onChange={(e) => setUserAnswer(e.target.value)}
            placeholder="在这里输入你的答案..."
            className="w-full h-24 p-3 border rounded-lg"
            disabled={isLoading}
          />
          <div className="text-sm text-gray-500 mt-1">
            {userAnswer.length} / 1000 字
          </div>
        </section>
        
        {/* 按钮 */}
        <section className="flex gap-4">
          <button
            onClick={() => {}}
            className="px-4 py-2 bg-yellow-500 text-white rounded"
          >
            ⏸ 暂停
          </button>
          <button
            onClick={handleSubmitAnswer}
            disabled={isLoading}
            className="px-4 py-2 bg-green-500 text-white rounded flex-1"
          >
            {isLoading ? '处理中...' : '✅ 提交答案'}
          </button>
          <button
            onClick={handleEndInterview}
            className="px-4 py-2 bg-red-500 text-white rounded"
          >
            🚪 结束面试
          </button>
        </section>
      </div>
    </div>
  )
}
```

### 4️⃣ 活体检测页 (/liveness/page.tsx)

**功能**: 面试前人脸校准和眨眼检测

**页面流程**:
```
START
  ⬇️
1. 请求摄像头权限
  ⬇️
2. 显示实时摄像头画面
  ├─ "请对准摄像头"
  ⬇️
3. 面部识别 (调用 face_gaze.py)
  ├─ 检测人脸位置
  ├─ 检测眼睛位置和方向
  ├─ 检测眨眼
  ⬇️
4. 校准完成
  ├─ "✓ 人脸识别成功"
  ├─ "✓ 眼睛校准成功"
  └─ "✓ 请眨眼 2 次以完成校准"
  ⬇️
5. 眨眼检测 (计数: 0/2)
  ├─ 用户眨眼 × 1
  ├─ 用户眨眼 × 2
  ⬇️
6. 校准完成
  ├─ "恭喜！校准成功"
  ⬇️
7. 进入面试
  └─ router.push('/interview')
```

**关键代码**:
```typescript
// liveness/page.tsx
'use client'

import { useEffect, useRef, useState } from 'react'
import { useRouter } from 'next/navigation'

export default function LivenessPage() {
  const router = useRouter()
  const videoRef = useRef<HTMLVideoElement>(null)
  const [status, setStatus] = useState('requesting')  // 'requesting' | 'detecting' | 'calibrating' | 'blinking' | 'success'
  const [blinkCount, setBlinkCount] = useState(0)
  const [faceDetected, setFaceDetected] = useState(false)
  
  useEffect(() => {
    // 1. 请求摄像头权限
    navigator.mediaDevices.getUserMedia({ video: true })
      .then(stream => {
        if (videoRef.current) {
          videoRef.current.srcObject = stream
          setStatus('detecting')
        }
      })
      .catch(err => {
        alert('无法访问摄像头: ' + err.message)
      })
  }, [])
  
  useEffect(() => {
    if (status !== 'detecting') return
    
    // 2. 调用后端 face_gaze.py 进行人脸检测和眨眼检测
    const detectLiveness = async () => {
      try {
        // 从 video 捕获帧
        if (videoRef.current) {
          const canvas = document.createElement('canvas')
          canvas.width = videoRef.current.videoWidth
          canvas.height = videoRef.current.videoHeight
          const ctx = canvas.getContext('2d')
          ctx.drawImage(videoRef.current, 0, 0)
          
          // 发送给后端识别
          const response = await fetch('/api/liveness/detect', {
            method: 'POST',
            headers: { 'Content-Type': 'application/json' },
            body: JSON.stringify({
              frame: canvas.toDataURL('image/jpeg')
            })
          })
          
          const result = await response.json()
          
          if (result.face_detected) {
            setFaceDetected(true)
            setStatus('calibrating')
          }
          
          if (result.blink_detected) {
            setBlinkCount(prev => prev + 1)
          }
          
          if (blinkCount >= 2) {
            setStatus('success')
            setTimeout(() => {
              router.push('/interview')
            }, 1000)
          }
        }
        
        // 持续检测
        requestAnimationFrame(detectLiveness)
      } catch (err) {
        console.error('检测失败:', err)
      }
    }
    
    detectLiveness()
  }, [status, blinkCount])
  
  return (
    <div className="flex flex-col items-center justify-center min-h-screen">
      <h1 className="text-3xl font-bold mb-8">👤 活体检测</h1>
      
      <video
        ref={videoRef}
        autoPlay
        playsInline
        className="w-96 h-72 rounded-lg bg-black mb-8"
      />
      
      {status === 'requesting' && <p>📹 请求摄像头权限...</p>}
      {status === 'detecting' && <p>👁️ 检测人脸...</p>}
      {faceDetected && status === 'calibrating' && <p>✓ 人脸识别成功</p>}
      {status === 'blinking' && <p>👁️ 请眨眼 ({blinkCount}/2)</p>}
      {status === 'success' && <p>✅ 校准成功，准备进入面试...</p>}
    </div>
  )
}
```

```typescript
// report/page.tsx
'use client'

import { useEffect, useState } from 'react'
import { useSearchParams } from 'next/navigation'

interface Report {
  interview_id: string
  created_at: string
  round_type: string
  position: string
  difficulty: string
  overall_score: number
  questions_and_answers: any[]
  strengths: string[]
  weaknesses: string[]
  suggestions: string[]
}

export default function ReportPage() {
  const searchParams = useSearchParams()
  const [reports, setReports] = useState<Report[]>([])
  const [selectedReport, setSelectedReport] = useState<Report | null>(null)
  const [isLoading, setIsLoading] = useState(true)
  
  useEffect(() => {
    // 获取所有报告
    fetch('/api/reports')
      .then(res => res.json())
      .then(data => {
        setReports(data.reports)
        
        // 如果 URL 中有 reportPath 参数，选中该报告
        const reportPath = searchParams.get('reportPath')
```

## 🔌 Socket 连接管理 (lib/socket.ts)

```typescript
// socket.ts
import io, { Socket } from 'socket.io-client'

export class SocketClient {
  private static instance: SocketClient
  private socket: Socket | null = null
  private connecting: Promise<void> | null = null
  
  static getInstance(): SocketClient {
    if (!this.instance) {
      this.instance = new SocketClient()
    }
    return this.instance
  }
  
  async connect(): Promise<void> {
    if (this.socket?.connected) {
      return  // 已连接
    }
    
    if (this.connecting) {
      return this.connecting  // 正在连接，等待
    }
    
    this.connecting = new Promise((resolve, reject) => {
      const socket = io(process.env.NEXT_PUBLIC_BACKEND_URL || 'http://localhost:5000', {
        reconnection: true,
        reconnectionDelay: 1000,
        reconnectionDelayMax: 5000,
        reconnectionAttempts: 5,
        transports: ['websocket', 'polling']  // WebSocket 优先，自动降级到 Polling
      })
      
      socket.on('connect', () => {
        console.log('✓ Socket 已连接')
        this.socket = socket
        this.connecting = null
        resolve()
      })
      
      socket.on('connect_error', (error) => {
        console.error('❌ Socket 连接错误:', error)
        reject(error)
      })
      
      this.socket = socket
    })
    
    return this.connecting
  }
  
  on(eventName: string, callback: (...args: any[]) => void) {
    if (this.socket) {
      this.socket.on(eventName, callback)
    }
  }
  
  emit(eventName: string, data?: any) {
    if (this.socket?.connected) {
      this.socket.emit(eventName, data)
    } else {
      console.warn('⚠️ Socket 未连接，无法发送:', eventName)
    }
  }
  
  disconnect() {
    if (this.socket) {
      this.socket.disconnect()
    }
  }
}
```

```typescript
// context/InterviewContext.tsx
import { createContext, useContext, useState } from 'react'

interface InterviewContextType {
  config: {round, position, difficulty}
  updateConfig: (config) => void
  chatHistory: any[]
  addMessage: (msg) => void
  currentQuestion: string
  setCurrentQuestion: (q) => void
}

const InterviewContext = createContext<InterviewContextType>({...})

export const useInterview = () => {
  return useContext(InterviewContext)
}

export function InterviewProvider({ children }) {
  const [config, setConfig] = useState(null)
  const [chatHistory, setChatHistory] = useState([])
  const [currentQuestion, setCurrentQuestion] = useState('')
  
  const value: InterviewContextType = {
    config,
    updateConfig: setConfig,
    chatHistory,
    addMessage: (msg) => setChatHistory(prev => [...prev, msg]),
    currentQuestion,
    setCurrentQuestion
  }
  
  return (
    <InterviewContext.Provider value={value}>
      {children}
    </InterviewContext.Provider>
  )
}

// 在 layout.tsx 中使用
export default function RootLayout({ children }) {
  return (
    <InterviewProvider>
      {children}
    </InterviewProvider>
  )
}
```
